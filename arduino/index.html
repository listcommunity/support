Usage

work in 'git bash' at Windows with Git installed or Linux OS

sh update.sh

usage()
{
    echo ""
    echo " usage:"
    echo ""
    echo "./update.sh "
    echo "    where  is one of:"
    echo "      --install-or-update     (does full installation or update.)"
    echo "      --remove 		        (removes all installed)"
    echo ""
    echo "example:"
    echo '    $ ./update.sh --install-or-update'
}
